# Games Menu scans the included shortcuts/ directory by default.
# Put .lnk and .url files there; every discovered shortcut appears in ALL.
# Optional artwork is matched by shortcut basename:
#
#   user-assets\<ShortcutBaseName>.png
#
# Example:
#   shortcuts\SnowRunner.url
#   user-assets\SnowRunner.png
#
# Optional advanced override:
# shortcutsDir=C:\Path\To\External Game Shortcuts

# Game metadata:
# item=<ID>|<Title>|<GlobalSubtitle>
#
# IDs are assigned automatically from shortcut launch identity and stored in:
#   state\items.json
#
# Per-card local overrides for title, description, image, folder, and order
# are also stored in state\items.json. They do not rename or modify shortcuts.
#
# Folder entries can use:
# folder=<folderId>|<DisplayName>|<MaxColumns>
# game=<ID>
# game=<ID>|<FolderSubtitle>
